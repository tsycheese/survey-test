\section{附录 A：核心数据模型定义}

\begin{minted}[linenos]{typescript}
// 用户认证模型
model User {
  id            String    @id @default(cuid())
  name          String?
  email         String    @unique
  emailVerified DateTime?
  image         String?
  password      String?
  accounts      Account[]
  sessions      Session[]
  surveys       Survey[]
  collaborators SurveyCollaborator[]
}

// 问卷模型
model Survey {
  id               String   @id @default(cuid())
  title            String
  description      String?
  published        Boolean  @default(false)
  shareToken       String?  @unique
  currentVersionId String?
  userId           String
  createdAt        DateTime @default(now())
  updatedAt        DateTime @updatedAt

  user          User                 @relation(fields: [userId], references: [id])
  questions     Question[]
  responses     Response[]
  versions      SurveyVersion[]
  collaborators SurveyCollaborator[]
  invites       SurveyInvite[]
  logs          SurveyLog[]
  views         SurveyView[]
}

// 题目模型
model Question {
  id       String @id @default(cuid())
  surveyId String
  type     String
  title    String
  required Boolean @default(false)
  order    Int
  config   Json?

  survey  Survey   @relation(fields: [surveyId], references: [id], onDelete: Cascade)
  options Option[]
  answers Answer[]
}

// 选项模型
model Option {
  id         String @id @default(cuid())
  questionId String
  label      String
  value      String?
  order      Int

  question Question @relation(fields: [questionId], references: [id], onDelete: Cascade)
}

// 回答模型
model Response {
  id         String   @id @default(cuid())
  surveyId   String
  versionId  String?
  createdAt  DateTime @default(now())

  survey  Survey       @relation(fields: [surveyId], references: [id], onDelete: Cascade)
  version SurveyVersion? @relation(fields: [versionId], references: [id])
  answers Answer[]
}

// 答案模型
model Answer {
  id         String @id @default(cuid())
  responseId String
  questionId String
  value      Json

  response Response @relation(fields: [responseId], references: [id], onDelete: Cascade)
  question Question @relation(fields: [questionId], references: [id])
}

// 版本模型
model SurveyVersion {
  id        String   @id @default(cuid())
  surveyId  String
  version   Int
  title     String
  questions Json
  createdAt DateTime @default(now())

  survey    Survey     @relation(fields: [surveyId], references: [id], onDelete: Cascade)
  responses Response[]
}
\end{minted}
\captionof{listing}{完整 Prisma Schema（核心部分）}
